CREATE OR REPLACE PROCEDURE FIN_A.SCH.CURATED_BRANCH_DATA_UPDATE_IN_DIM_ACCOUNT()
RETURNS VARCHAR
LANGUAGE SQL
EXECUTE AS OWNER
AS '
    BEGIN
        
        UPDATE FIN_A.SCH.DIM_ACCOUNT A
        SET A.ACCOUNT_STATUS=B.BRANCH_STATUS
        FROM FIN_A.SCH.DIM_BRANCH B
        WHERE A.BRANCH_CODE =B.BRANCH_CODE AND B.BRANCH_STATUS<>A.ACCOUNT_STATUS;

    END;
';
