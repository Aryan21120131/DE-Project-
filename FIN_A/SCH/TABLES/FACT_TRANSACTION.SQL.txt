create or replace TABLE FIN_A.SCH.FACT_TRANSACTION (
	TRANSACTION_ID VARCHAR(50) NOT NULL COMMENT 'Unique transaction refference number',
	ACCOUNT_ID VARCHAR(100) NOT NULL COMMENT 'Refferences DIM_ACCOUNT.account_id',
	TRANSACTION_DATE TIMESTAMP_NTZ(9) NOT NULL COMMENT 'Date & time of transaction',
	TRANSACTION_TYPE VARCHAR(10) NOT NULL COMMENT 'DEBIT/CREDI',
	TRANSACTION_AMOUNT NUMBER(18,2) NOT NULL COMMENT 'Amount transfered',
	TRANSACTION_MODE VARCHAR(20) NOT NULL COMMENT 'UPI/CASH/NEFT/IMPS/CARD',
	COUNTERPARTY_ACCOUNT VARCHAR(30) COMMENT 'Other account involved',
	COUNTERPARTY_BANK VARCHAR(100) COMMENT 'Other bank name',
	TRANSACTION_STATUS VARCHAR(20) NOT NULL COMMENT 'SUCESS/FAILED',
	NARRATION VARCHAR(300) COMMENT 'Transaction description',
	primary key (TRANSACTION_ID)
);
