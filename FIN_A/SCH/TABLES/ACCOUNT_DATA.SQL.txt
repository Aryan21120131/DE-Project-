create or replace TABLE FIN_A.SCH.ACCOUNT_DATA (
	ACCOUNT_NUMBER VARCHAR(30) NOT NULL COMMENT 'Business account number',
	PAN_NUMBER VARCHAR(10) NOT NULL COMMENT 'Customer PAN',
	ACCOUNT_TYPE VARCHAR(20) NOT NULL COMMENT 'SAVINGS/CURRENT',
	BRANCH_CODE VARCHAR(20) NOT NULL COMMENT 'Branch where account opened',
	CURRENCY VARCHAR(10) NOT NULL COMMENT 'INR/USD',
	OPENING_DATE DATE NOT NULL COMMENT 'Account opening date',
	OPENING_BALANCE NUMBER(18,2) NOT NULL COMMENT 'Initial balance',
	ACCOUNT_STATUS VARCHAR(20) NOT NULL COMMENT 'ACTIVE/DORMANT/CLOSED'
);
