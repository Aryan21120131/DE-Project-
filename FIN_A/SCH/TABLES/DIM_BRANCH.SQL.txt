create or replace TABLE FIN_A.SCH.DIM_BRANCH (
	BRANCH_CODE VARCHAR(20) NOT NULL COMMENT 'Unique business identifier for branch',
	BRANCH_NAME VARCHAR(150) NOT NULL COMMENT 'Official branch name',
	CITY VARCHAR(100) NOT NULL COMMENT 'City where branch is located',
	STATE VARCHAR(100) NOT NULL COMMENT 'State',
	COUNTRY VARCHAR(100) NOT NULL COMMENT 'Country',
	IFSC_CODE VARCHAR(15) NOT NULL COMMENT 'IFSC/routing code',
	BRANCH_STATUS VARCHAR(20) NOT NULL COMMENT 'ACTIVE/INACTIVE',
	unique (IFSC_CODE),
	primary key (BRANCH_CODE)
);
