create or replace TABLE FIN_A.TRA.DIM_CUSTOMER (
	CUSTOMER_ID NUMBER(38,0) NOT NULL COMMENT 'Surrogate key for customer',
	CUSTOMER_NAME VARCHAR(150) NOT NULL COMMENT 'Full name of customer',
	CUSTOMER_TYPE VARCHAR(20) NOT NULL COMMENT 'INDIVIDUAL/CORPORATE',
	DATE_OF_BIRTH DATE COMMENT 'DOB(mandatory for individuals)',
	GENDER VARCHAR(10) COMMENT 'MALE/FEMALE/OTHER',
	PAN_NUMBER VARCHAR(10) NOT NULL COMMENT 'Unique PAN identifier',
	AADHAAR_NUMBER VARCHAR(12) COMMENT 'Aadhaar number',
	EMAIL VARCHAR(150) COMMENT 'Email address',
	MOBILE_NUMBER VARCHAR(15) NOT NULL COMMENT 'Contact number',
	ADDRESS VARCHAR(300) COMMENT 'Residential address',
	CITY VARCHAR(100) COMMENT 'City',
	STATE VARCHAR(100) COMMENT 'State',
	COUNTRY VARCHAR(100) NOT NULL COMMENT 'Country',
	RISK_CATEGORY VARCHAR(20) NOT NULL COMMENT 'LOW/MEDIUM/HIGH',
	ACTN_CD VARCHAR(1) NOT NULL COMMENT 'ACTON CODE',
	unique (PAN_NUMBER),
	unique (AADHAAR_NUMBER),
	primary key (CUSTOMER_ID)
);
