create or replace TABLE FIN_A.TRA.DIM_ACCOUNT (
	ACCOUNT_ID VARCHAR(100) NOT NULL,
	CUSTOMER_ID NUMBER(38,0) NOT NULL COMMENT 'Refferences DIM_CUSTOMER.CUSTOMER_ID',
	ACCOUNT_NUMBER VARCHAR(30) NOT NULL COMMENT 'Bank account number',
	BRANCH_CODE VARCHAR(20) NOT NULL COMMENT 'Refferences DIM_BRANCH.BRANCH_CODE',
	ACCOUNT_TYPE VARCHAR(20) NOT NULL COMMENT 'CURRENT/SAVINGS',
	CURRENCY VARCHAR(10) NOT NULL COMMENT 'ISO currency code(INR,USD)',
	OPENING_DATE DATE NOT NULL COMMENT 'Account opening date',
	OPENING_BALANCE NUMBER(18,2) NOT NULL COMMENT 'Initial balance at opening',
	ACCOUNT_STATUS VARCHAR(20) NOT NULL COMMENT 'ACTIVE/INACTIVE',
	ACTN_CD VARCHAR(1) NOT NULL COMMENT 'ACTION CODE',
	unique (ACCOUNT_NUMBER),
	primary key (ACCOUNT_ID)
);
